export default function selectionSort(arr) {
    let i = 0
    while (i < arr.length){
        let minIndex = i;
        for (let j = i; j < arr.length; j++) {
            if(arr[j]<arr[minIndex]){
                minIndex = j
            }
        }
        [arr[i],arr[minIndex]] = [arr[minIndex],arr[i]]
        i++;
    }
    return arr
}
